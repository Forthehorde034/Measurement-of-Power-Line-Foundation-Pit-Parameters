
### 根据pyproject.toml生成requirements-xxx.txt锁定文件（仅开发）
uv pip compile pyproject.toml --group dev --output-file=requirements-dev.txt

### 生成最小化依赖（仅打包工具）
uv pip compile --group build --output-file=requirements-build.txt pyproject.toml

### 生成带哈希的生产依赖（部署用）
uv pip compile pyproject.toml --output-file=requirements.txt

### 安装依赖
uv pip install -r requirements-dev.txt

uv pip install -r requirements-build.txt
